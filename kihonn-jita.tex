  % 测量文本宽度

  % 先判断是否 <= 1\zw
  \dim_compare:nTF { \box_wd:N \l_tmpa_box <= 1\zw }
    { #2 } % 太短，直接原样输出
    {
      % 否则再执行原来的逻辑
      \dim_compare:nTF { \box_wd:N \l_tmpa_box < #1 }
        { \jidori{#1}{#2} }
        { #2 }
    }
 }
